Using modules with require();

                

                
These are modules.

                    
var http = require('http'),
    url = require('url'),
    fs = require('fs'),
    qs = require('querystring');
                    

                    
But you can now use the modules you've included in your dependancies, so check the docs for the actual module name.

                    
var _ = require('underscore'); //returns methods attached to 'export'.
                    

                

            


            

                
exports Object

                

                
require() returns a reference to the exports object, with methods from the module attached.

                
var circle = require('./circle.js');
console.log('The area of a circle of radius 4 is '
           + circle.area(4));



The actual file:


//Local to module, or 'private'
var PI = Math.PI;
//public function, part of the module.
exports.area = function (r) {
  return PI * r * r;
};

exports.circumference = function (r) {
  return 2 * PI * r;
};
